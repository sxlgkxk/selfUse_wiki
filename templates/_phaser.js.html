  • API
    • preload, create, update
    • game=Phaser.Game(config)
      • config: type:Phaser.AUTO
        • width, height, scene:{preload:preload}
    • preload
      • baseURL: this.load.baseURL='static/media/image/piskel/'
      • loadImage: this.load.image('player', 'human/player/head1.png');
      • loadSprite: this.load.spritesheet('','',{frameWidth:32, frameHeight:48})
    • create
      • addImage: this.add.image(0,0,'sky').setOrigin(0,0)
      • physics staticGroup: platform=this.physics.add.staticGroup()
        • init node(with scale): platform.create(200,200,'ground').setScale(2).refreshBody()
      • physics dynamic: player=this.physics.add.sprite(1,1,'s')
        • set attributes: player.setBounce(0.2), .setCollideWorldBounds(true)
      • control: keys=this.input.keyboard.createCursorKeys();
        • if keys.left.isDown
      • scale: this.scale.on('resize',resize)
